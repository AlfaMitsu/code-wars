    function check(a, x) {
      return a.includes(x);
    }
    
    // Examples
    console.log(check([1, 2, 3], 2)); // true
    console.log(check(['apple', 'banana', 'cherry'], 'banana')); // true
    console.log(check([1, 2, 3], 4)); // false
    console.log(check(['apple', 'banana', 'cherry'], 'orange')); // false
